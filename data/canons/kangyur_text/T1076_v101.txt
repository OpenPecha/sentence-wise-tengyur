པྲ་དཀྵཱ་རཏྣ་ཏྲ་ཡཱ་ནཱ་མ་དྷཱ་ར་ཎཱི། བོད་སྐད་དུ། འཕགས་པ་དཀོན་མཆོག་གི་རྟེན་ལ་བསྐོར་བ་བྱ་བའི་གཟུངས་ཞེས་བྱ་བ། བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ཤཱཀྱ་ཐུབ་པ་ལ་ཕྱག་འཚལ་ལོ། །ཨོཾ་ན་མོ་

ད་ཤ་དི་ཀ་ཏྲི་ཀཱ་ལ་སརྦ་རཏྣ་ཏྲ་ཡཱ་ཡ། མ་མ་པྲ་དཀྵཱ་སུ་པྲ་དཀྵཱ། སརྦ་བཱ་པཾ་པི་ཤོ་དྷ་ནི་སྭཱ་ཧཱ། གཟུངས་འདི་ལན་ཅིག་བརྗོད་པས་ཕྱོགས་བཅུ་དུས་གསུམ་གྱི་དཀོན་མཆོག་གསུམ་ལ་ཕྱག་འཚལ་ཞིང་བསྐོར་བ་བྱས་པར་འགྱུར་རོ། །ཐོག་མ་མེད་པ་
ནས་བསགས་པའི་སྡིག་པ་ཐམས་ཅད་དག་པར་འགྱུར་རོ། །སངས་རྒྱས་ཀྱི་བྱང་ཆུབ་མྱུར་དུ་ཐོབ་པར་འགྱུར་རོ། །དགྲ་དང་བགེགས་ཀྱིས་གླགས་རྙེད་པར་མི་འགྱུར་རོ། །ནད་དང་གདོན་ཐམས་ཅད་ལས་ཐར་བར་འགྱུར་རོ། །བསྐོར་བའི་གཟུངས་རྫོགས་
