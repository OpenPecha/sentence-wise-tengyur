སངས་རྒྱས་ཚུལ་དོན་གང་ཡིན་པ།། 
སྐྱེ་མེད་འབྱུང་བ་མེད་པ་ཡིན།། 
བྱམས་པས་སེམས་ཅན་ཁྱབ་བྱེད་ཀྱང་།། 
སེམས་ཅན་དམིགས་སུ་ཡོད་མ་ཡིན།། 
སེམས་ཅན་དོན་ནི་རྗེས་རྟོགས་པ།། 
དེ་ནི་མིང་གིས་བསྟན་པ་ཡིན།། 
འདུས་བྱས་ཀུན་ལ་རྣམ་སེམས་པའི།། 
བརྩོན་འགྲུས་རྩོམ་ལ་གནས་གྱུར་ཅིང་།། 
འཇིག་རྟེན་སྟོང་པར་རིག་ནས་ནི།། 
བྱང་ཆུབ་དམ་པ་རྟོགས་པར་བྱེད།། 
དེ་ནི་བསམ་གཏན་སྒོམ་བྱེད་ཀྱང་།། 
དེ་ལ་རྟེན་པ་ཡོད་མིན་ཞིང་།། 
གནས་མེད་དམིགས་པ་ཡོད་མིན་པ།། 
དེ་ལྟར་མཁས་པ་སྒོམ་པར་བྱེད།། 
བག་ཆགས་ལྟ་བའི་ཉོན་མོངས་རྣམས།། 
ཤེས་རབ་མཚོན་གྱིས་གཅོད་བྱེད་ཀྱང་།། 
ཆོས་ཀྱི་དབྱིངས་ལ་དབྱེ་བ་དང་།། 
གཅོད་པའི་རློམ་སེམས་ཡོད་མ་ཡིན།།
