དེས་ནི་སངས་རྒྱས་ཉིད་ཀྱང་འཐོབ།། 
གང་ཞིག་མིང་བརྒྱ་རྩ་བརྒྱད་འདོན།། 
མི་དེ་ཡོན་ཏན་ཡོངས་རྫོགས་ཤིང་།། 
དེ་བཞིན་གཤེགས་ཚུལ་ཆེ་བ་ཡི།། 
ཡེ་ཤེས་མ་ལུས་སྨྲ་བར་ནུས།། 
དེ་ལྟར་ཐུབ་མཆོག་སྲས་པོ་ཡོན་ཏན་བརྒྱ་ཡི་སྤོབས་ལས་ངེས་པར་བྱུང་གྱུར་ཅིང་།། 
ཡོན་ཏན་སྣ་ཚོགས་འབྱུང་དང་འཇིག་རྟེན་ཡན་ཆད་དག་གིས་ཡོན་ཏན་རྒྱས་རྣམས་ལ།། 
བསྟོད་པ་ཆེན་པོ་རྣམས་བྱས་རྒྱལ་བའི་སྲས་ལ་བསྟོད་པས་ཉིན་རེ་བསྟོད་བྱས་ན།། 
སྟོབས་བཅུ་མངའ་བའི་སྐུར་འགྱུར་སྣང་བ་ཐོབ་ཅིང་རྣམ་དག་ཡོན་ཏན་འབྱུང་གནས་འགྱུར།། 
འཕགས་པ་ཀུན་ཏུ་བཟང་པོའི་མཚན་བརྒྱ་རྩ་བརྒྱད་པ་གཟུངས་སྔགས་དང་བཅས་པ་རྫོགས་སོ།།
